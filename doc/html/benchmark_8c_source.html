109 #define TIME_AND_TSC( TITLE, CODE ) \
110 do { \
111  unsigned long i, j, tsc; \
112  \
113  mbedtls_printf( HEADER_FORMAT, TITLE ); \
114  fflush( stdout ); \
115  \
116  mbedtls_set_alarm( 1 ); \
117  for( i = 1; ! mbedtls_timing_alarmed; i++ ) \
118  { \
119  CODE; \
120  } \
121  \
122  tsc = mbedtls_timing_hardclock(); \
123  for( j = 0; j < 1024; j++ ) \
124  { \
125  CODE; \
126  } \
127  \
128  mbedtls_printf( "%9lu Kb/s, %9lu cycles/byte\n", \
129  i * BUFSIZE / 1024, \
130  ( mbedtls_timing_hardclock() - tsc ) / ( j * BUFSIZE ) ); \
131 } while( 0 )

163 #define TIME_PUBLIC( TITLE, TYPE, CODE ) \
164 do { \
165  unsigned long i; \
166  int ret; \
167  MEMORY_MEASURE_INIT; \
168  \
169  mbedtls_printf( HEADER_FORMAT, TITLE ); \
170  fflush( stdout ); \
171  mbedtls_set_alarm( 3 ); \
172  \
173  ret = 0; \
174  for( i = 1; ! mbedtls_timing_alarmed && ! ret ; i++ ) \
175  { \
176  CODE; \
177  } \
178  \
179  if( ret != 0 ) \
180  { \
181  PRINT_ERROR; \
182  } \
183  else \
184  { \
185  mbedtls_printf( "%6lu " TYPE "/s", i / 3 ); \
186  MEMORY_MEASURE_PRINT( sizeof( TYPE ) + 1 ); \
187  mbedtls_printf( "\n" ); \
188  } \
189 } while( 0 )

191 static int myrand( void *rng_state, unsigned char *output, size_t len )
192 {
193  size_t use_len;
194  int rnd;
195 
196  if( rng_state != NULL )
197  rng_state = NULL;
198 
199  while( len > 0 )
200  {
201  use_len = len;
202  if( use_len > sizeof(int) )
203  use_len = sizeof(int);
204 
205  rnd = rand();
206  memcpy( output, &rnd, use_len );
207  output += use_len;
208  len -= use_len;
209  }
210 
211  return( 0 );
212 }
