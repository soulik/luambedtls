51 static int generic_wrapper( const mbedtls_md_info_t *md_info, char *filename, unsigned char *sum )
52 {
53  int ret = mbedtls_md_file( md_info, filename, sum );
54 
55  if( ret == 1 )
56  mbedtls_fprintf( stderr, "failed to open: %s\n", filename );
57 
58  if( ret == 2 )
59  mbedtls_fprintf( stderr, "failed to read: %s\n", filename );
60 
61  return( ret );
62 }

64 static int generic_print( const mbedtls_md_info_t *md_info, char *filename )
65 {
66  int i;
67  unsigned char sum[MBEDTLS_MD_MAX_SIZE];
68 
69  if( generic_wrapper( md_info, filename, sum ) != 0 )
70  return( 1 );
71 
72  for( i = 0; i < mbedtls_md_get_size( md_info ); i++ )
73  mbedtls_printf( "%02x", sum[i] );
74 
75  mbedtls_printf( " %s\n", filename );
76  return( 0 );
77 }

79 static int generic_check( const mbedtls_md_info_t *md_info, char *filename )
80 {
81  int i;
82  size_t n;
83  FILE *f;
84  int nb_err1, nb_err2;
85  int nb_tot1, nb_tot2;
86  unsigned char sum[MBEDTLS_MD_MAX_SIZE];
87  char buf[MBEDTLS_MD_MAX_SIZE * 2 + 1], line[1024];
88  char diff;
89 
90  if( ( f = fopen( filename, "rb" ) ) == NULL )
91  {
92  mbedtls_printf( "failed to open: %s\n", filename );
93  return( 1 );
94  }
95 
96  nb_err1 = nb_err2 = 0;
97  nb_tot1 = nb_tot2 = 0;
98 
99  memset( line, 0, sizeof( line ) );
100 
101  n = sizeof( line );
102 
103  while( fgets( line, (int) n - 1, f ) != NULL )
104  {
105  n = strlen( line );
106 
107  if( n < (size_t) 2 * mbedtls_md_get_size( md_info ) + 4 )
108  {
109  mbedtls_printf("No '%s' hash found on line.\n", mbedtls_md_get_name( md_info ));
110  continue;
111  }
112 
113  if( line[2 * mbedtls_md_get_size( md_info )] != ' ' || line[2 * mbedtls_md_get_size( md_info ) + 1] != ' ' )
114  {
115  mbedtls_printf("No '%s' hash found on line.\n", mbedtls_md_get_name( md_info ));
116  continue;
117  }
118 
119  if( line[n - 1] == '\n' ) { n--; line[n] = '\0'; }
120  if( line[n - 1] == '\r' ) { n--; line[n] = '\0'; }
121 
122  nb_tot1++;
123 
124  if( generic_wrapper( md_info, line + 2 + 2 * mbedtls_md_get_size( md_info ), sum ) != 0 )
125  {
126  nb_err1++;
127  continue;
128  }
129 
130  nb_tot2++;
131 
132  for( i = 0; i < mbedtls_md_get_size( md_info ); i++ )
133  sprintf( buf + i * 2, "%02x", sum[i] );
134 
135  /* Use constant-time buffer comparison */
136  diff = 0;
137  for( i = 0; i < 2 * mbedtls_md_get_size( md_info ); i++ )
138  diff |= line[i] ^ buf[i];
139 
140  if( diff != 0 )
141  {
142  nb_err2++;
143  mbedtls_fprintf( stderr, "wrong checksum: %s\n", line + 66 );
144  }
145 
146  n = sizeof( line );
147  }
148 
149  if( nb_err1 != 0 )
150  {
151  mbedtls_printf( "WARNING: %d (out of %d) input files could "
152  "not be read\n", nb_err1, nb_tot1 );
153  }
154 
155  if( nb_err2 != 0 )
156  {
157  mbedtls_printf( "WARNING: %d (out of %d) computed checksums did "
158  "not match\n", nb_err2, nb_tot2 );
159  }
160 
161  fclose( f );
162 
163  return( nb_err1 != 0 || nb_err2 != 0 );
164 }

166 int main( int argc, char *argv[] )
167 {
168  int ret, i;
169  const mbedtls_md_info_t *md_info;
170  mbedtls_md_context_t md_ctx;
171 
172  mbedtls_md_init( &md_ctx );
173 
174  if( argc == 1 )
175  {
176  const int *list;
177 
178  mbedtls_printf( "print mode: generic_sum <mbedtls_md> <file> <file> ...\n" );
179  mbedtls_printf( "check mode: generic_sum <mbedtls_md> -c <checksum file>\n" );
180 
181  mbedtls_printf( "\nAvailable message digests:\n" );
182  list = mbedtls_md_list();
183  while( *list )
184  {
185  md_info = mbedtls_md_info_from_type( *list );
186  mbedtls_printf( " %s\n", mbedtls_md_get_name( md_info ) );
187  list++;
188  }
189 
190 #if defined(_WIN32)
191  mbedtls_printf( "\n Press Enter to exit this program.\n" );
192  fflush( stdout ); getchar();
193 #endif
194 
195  return( 1 );
196  }
197 
198  /*
199  * Read the MD from the command line
200  */
201  md_info = mbedtls_md_info_from_string( argv[1] );
202  if( md_info == NULL )
203  {
204  mbedtls_fprintf( stderr, "Message Digest '%s' not found\n", argv[1] );
205  return( 1 );
206  }
207  if( mbedtls_md_setup( &md_ctx, md_info, 0 ) )
208  {
209  mbedtls_fprintf( stderr, "Failed to initialize context.\n" );
210  return( 1 );
211  }
212 
213  ret = 0;
214  if( argc == 4 && strcmp( "-c", argv[2] ) == 0 )
215  {
216  ret |= generic_check( md_info, argv[3] );
217  goto exit;
218  }
219 
220  for( i = 2; i < argc; i++ )
221  ret |= generic_print( md_info, argv[i] );
222 
223 exit:
224  mbedtls_md_free( &md_ctx );
225 
226  return( ret );
227 }
