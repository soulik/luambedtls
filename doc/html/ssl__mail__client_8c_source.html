165 static int do_handshake( mbedtls_ssl_context *ssl )
166 {
167  int ret;
168  uint32_t flags;
169  unsigned char buf[1024];
170  memset(buf, 0, 1024);
171 
172  /*
173  * 4. Handshake
174  */
175  mbedtls_printf( " . Performing the SSL/TLS handshake..." );
176  fflush( stdout );
177 
178  while( ( ret = mbedtls_ssl_handshake( ssl ) ) != 0 )
179  {
180  if( ret != MBEDTLS_ERR_SSL_WANT_READ && ret != MBEDTLS_ERR_SSL_WANT_WRITE )
181  {
182 #if defined(MBEDTLS_ERROR_C)
183  mbedtls_strerror( ret, (char *) buf, 1024 );
184 #endif
185  mbedtls_printf( " failed\n ! mbedtls_ssl_handshake returned %d: %s\n\n", ret, buf );
186  return( -1 );
187  }
188  }
189 
190  mbedtls_printf( " ok\n [ Ciphersuite is %s ]\n",
191  mbedtls_ssl_get_ciphersuite( ssl ) );
192 
193  /*
194  * 5. Verify the server certificate
195  */
196  mbedtls_printf( " . Verifying peer X.509 certificate..." );
197 
198  /* In real life, we probably want to bail out when ret != 0 */
199  if( ( flags = mbedtls_ssl_get_verify_result( ssl ) ) != 0 )
200  {
201  char vrfy_buf[512];
202 
203  mbedtls_printf( " failed\n" );
204 
205  mbedtls_x509_crt_verify_info( vrfy_buf, sizeof( vrfy_buf ), " ! ", flags );
206 
207  mbedtls_printf( "%s\n", vrfy_buf );
208  }
209  else
210  mbedtls_printf( " ok\n" );
211 
212  mbedtls_printf( " . Peer certificate information ...\n" );
213  mbedtls_x509_crt_info( (char *) buf, sizeof( buf ) - 1, " ",
214  mbedtls_ssl_get_peer_cert( ssl ) );
215  mbedtls_printf( "%s\n", buf );
216 
217  return( 0 );
218 }

237 static int write_ssl_and_get_response( mbedtls_ssl_context *ssl, unsigned char *buf, size_t len )
238 {
239  int ret;
240  unsigned char data[128];
241  char code[4];
242  size_t i, idx = 0;
243 
244  mbedtls_printf("\n%s", buf);
245  while( len && ( ret = mbedtls_ssl_write( ssl, buf, len ) ) <= 0 )
246  {
247  if( ret != MBEDTLS_ERR_SSL_WANT_READ && ret != MBEDTLS_ERR_SSL_WANT_WRITE )
248  {
249  mbedtls_printf( " failed\n ! mbedtls_ssl_write returned %d\n\n", ret );
250  return -1;
251  }
252  }
253 
254  do
255  {
256  len = sizeof( data ) - 1;
257  memset( data, 0, sizeof( data ) );
258  ret = mbedtls_ssl_read( ssl, data, len );
259 
260  if( ret == MBEDTLS_ERR_SSL_WANT_READ || ret == MBEDTLS_ERR_SSL_WANT_WRITE )
261  continue;
262 
263  if( ret == MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY )
264  return -1;
265 
266  if( ret <= 0 )
267  {
268  mbedtls_printf( "failed\n ! mbedtls_ssl_read returned %d\n\n", ret );
269  return -1;
270  }
271 
272  mbedtls_printf("\n%s", data);
273  len = ret;
274  for( i = 0; i < len; i++ )
275  {
276  if( data[i] != '\n' )
277  {
278  if( idx < 4 )
279  code[ idx++ ] = data[i];
280  continue;
281  }
282 
283  if( idx == 4 && code[0] >= '0' && code[0] <= '9' && code[3] == ' ' )
284  {
285  code[3] = '\0';
286  return atoi( code );
287  }
288 
289  idx = 0;
290  }
291  }
292  while( 1 );
293 }

295 static int write_and_get_response( mbedtls_net_context *sock_fd, unsigned char *buf, size_t len )
296 {
297  int ret;
298  unsigned char data[128];
299  char code[4];
300  size_t i, idx = 0;
301 
302  mbedtls_printf("\n%s", buf);
303  if( len && ( ret = mbedtls_net_send( sock_fd, buf, len ) ) <= 0 )
304  {
305  mbedtls_printf( " failed\n ! mbedtls_ssl_write returned %d\n\n", ret );
306  return -1;
307  }
308 
309  do
310  {
311  len = sizeof( data ) - 1;
312  memset( data, 0, sizeof( data ) );
313  ret = mbedtls_net_recv( sock_fd, data, len );
314 
315  if( ret <= 0 )
316  {
317  mbedtls_printf( "failed\n ! read returned %d\n\n", ret );
318  return -1;
319  }
320 
321  data[len] = '\0';
322  mbedtls_printf("\n%s", data);
323  len = ret;
324  for( i = 0; i < len; i++ )
325  {
326  if( data[i] != '\n' )
327  {
328  if( idx < 4 )
329  code[ idx++ ] = data[i];
330  continue;
331  }
332 
333  if( idx == 4 && code[0] >= '0' && code[0] <= '9' && code[3] == ' ' )
334  {
335  code[3] = '\0';
336  return atoi( code );
337  }
338 
339  idx = 0;
340  }
341  }
342  while( 1 );
343 }
