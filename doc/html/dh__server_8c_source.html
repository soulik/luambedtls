61  mbedtls_printf("MBEDTLS_AES_C and/or MBEDTLS_DHM_C and/or MBEDTLS_ENTROPY_C "
62  "and/or MBEDTLS_NET_C and/or MBEDTLS_RSA_C and/or "
63  "MBEDTLS_SHA256_C and/or MBEDTLS_FS_IO and/or "

68 int main( void )
69 {
70  FILE *f;
71 
72  int ret;
73  size_t n, buflen;
74  mbedtls_net_context listen_fd, client_fd;
75 
76  unsigned char buf[2048];
77  unsigned char hash[20];
78  unsigned char buf2[2];
79  const char *pers = "dh_server";
80 
81  mbedtls_entropy_context entropy;
82  mbedtls_ctr_drbg_context ctr_drbg;
83  mbedtls_rsa_context rsa;
84  mbedtls_dhm_context dhm;
85  mbedtls_aes_context aes;
86 
87  mbedtls_net_init( &listen_fd );
88  mbedtls_net_init( &client_fd );
89  mbedtls_rsa_init( &rsa, MBEDTLS_RSA_PKCS_V15, MBEDTLS_MD_SHA256 );
90  mbedtls_dhm_init( &dhm );
91  mbedtls_aes_init( &aes );
92  mbedtls_ctr_drbg_init( &ctr_drbg );
93 
94  /*
95  * 1. Setup the RNG
96  */
97  mbedtls_printf( "\n . Seeding the random number generator" );
98  fflush( stdout );
99 
100  mbedtls_entropy_init( &entropy );
101  if( ( ret = mbedtls_ctr_drbg_seed( &ctr_drbg, mbedtls_entropy_func, &entropy,
102  (const unsigned char *) pers,
103  strlen( pers ) ) ) != 0 )
104  {
105  mbedtls_printf( " failed\n ! mbedtls_ctr_drbg_seed returned %d\n", ret );
106  goto exit;
107  }
108 
109  /*
110  * 2a. Read the server's private RSA key
111  */
112  mbedtls_printf( "\n . Reading private key from rsa_priv.txt" );
113  fflush( stdout );
114 
115  if( ( f = fopen( "rsa_priv.txt", "rb" ) ) == NULL )
116  {
117  ret = 1;
118  mbedtls_printf( " failed\n ! Could not open rsa_priv.txt\n" \
119  " ! Please run rsa_genkey first\n\n" );
120  goto exit;
121  }
122 
123  mbedtls_rsa_init( &rsa, MBEDTLS_RSA_PKCS_V15, 0 );
124 
125  if( ( ret = mbedtls_mpi_read_file( &rsa.N , 16, f ) ) != 0 ||
126  ( ret = mbedtls_mpi_read_file( &rsa.E , 16, f ) ) != 0 ||
127  ( ret = mbedtls_mpi_read_file( &rsa.D , 16, f ) ) != 0 ||
128  ( ret = mbedtls_mpi_read_file( &rsa.P , 16, f ) ) != 0 ||
129  ( ret = mbedtls_mpi_read_file( &rsa.Q , 16, f ) ) != 0 ||
130  ( ret = mbedtls_mpi_read_file( &rsa.DP, 16, f ) ) != 0 ||
131  ( ret = mbedtls_mpi_read_file( &rsa.DQ, 16, f ) ) != 0 ||
132  ( ret = mbedtls_mpi_read_file( &rsa.QP, 16, f ) ) != 0 )
133  {
134  mbedtls_printf( " failed\n ! mbedtls_mpi_read_file returned %d\n\n", ret );
135  goto exit;
136  }
137 
138  rsa.len = ( mbedtls_mpi_bitlen( &rsa.N ) + 7 ) >> 3;
139 
140  fclose( f );
141 
142  /*
143  * 2b. Get the DHM modulus and generator
144  */
145  mbedtls_printf( "\n . Reading DH parameters from dh_prime.txt" );
146  fflush( stdout );
147 
148  if( ( f = fopen( "dh_prime.txt", "rb" ) ) == NULL )
149  {
150  ret = 1;
151  mbedtls_printf( " failed\n ! Could not open dh_prime.txt\n" \
152  " ! Please run dh_genprime first\n\n" );
153  goto exit;
154  }
155 
156  if( mbedtls_mpi_read_file( &dhm.P, 16, f ) != 0 ||
157  mbedtls_mpi_read_file( &dhm.G, 16, f ) != 0 )
158  {
159  mbedtls_printf( " failed\n ! Invalid DH parameter file\n\n" );
160  goto exit;
161  }
162 
163  fclose( f );
164 
165  /*
166  * 3. Wait for a client to connect
167  */
168  mbedtls_printf( "\n . Waiting for a remote connection" );
169  fflush( stdout );
170 
171  if( ( ret = mbedtls_net_bind( &listen_fd, NULL, SERVER_PORT, MBEDTLS_NET_PROTO_TCP ) ) != 0 )
172  {
173  mbedtls_printf( " failed\n ! mbedtls_net_bind returned %d\n\n", ret );
174  goto exit;
175  }
176 
177  if( ( ret = mbedtls_net_accept( &listen_fd, &client_fd,
178  NULL, 0, NULL ) ) != 0 )
179  {
180  mbedtls_printf( " failed\n ! mbedtls_net_accept returned %d\n\n", ret );
181  goto exit;
182  }
183 
184  /*
185  * 4. Setup the DH parameters (P,G,Ys)
186  */
187  mbedtls_printf( "\n . Sending the server's DH parameters" );
188  fflush( stdout );
189 
190  memset( buf, 0, sizeof( buf ) );
191 
192  if( ( ret = mbedtls_dhm_make_params( &dhm, (int) mbedtls_mpi_size( &dhm.P ), buf, &n,
193  mbedtls_ctr_drbg_random, &ctr_drbg ) ) != 0 )
194  {
195  mbedtls_printf( " failed\n ! mbedtls_dhm_make_params returned %d\n\n", ret );
196  goto exit;
197  }
198 
199  /*
200  * 5. Sign the parameters and send them
201  */
202  mbedtls_sha1( buf, n, hash );
203 
204  buf[n ] = (unsigned char)( rsa.len >> 8 );
205  buf[n + 1] = (unsigned char)( rsa.len );
206 
207  if( ( ret = mbedtls_rsa_pkcs1_sign( &rsa, NULL, NULL, MBEDTLS_RSA_PRIVATE, MBEDTLS_MD_SHA256,
208  0, hash, buf + n + 2 ) ) != 0 )
209  {
210  mbedtls_printf( " failed\n ! mbedtls_rsa_pkcs1_sign returned %d\n\n", ret );
211  goto exit;
212  }
213 
214  buflen = n + 2 + rsa.len;
215  buf2[0] = (unsigned char)( buflen >> 8 );
216  buf2[1] = (unsigned char)( buflen );
217 
218  if( ( ret = mbedtls_net_send( &client_fd, buf2, 2 ) ) != 2 ||
219  ( ret = mbedtls_net_send( &client_fd, buf, buflen ) ) != (int) buflen )
220  {
221  mbedtls_printf( " failed\n ! mbedtls_net_send returned %d\n\n", ret );
222  goto exit;
223  }
224 
225  /*
226  * 6. Get the client's public value: Yc = G ^ Xc mod P
227  */
228  mbedtls_printf( "\n . Receiving the client's public value" );
229  fflush( stdout );
230 
231  memset( buf, 0, sizeof( buf ) );
232 
233  if( ( ret = mbedtls_net_recv( &client_fd, buf, n ) ) != (int) n )
234  {
235  mbedtls_printf( " failed\n ! mbedtls_net_recv returned %d\n\n", ret );
236  goto exit;
237  }
238 
239  if( ( ret = mbedtls_dhm_read_public( &dhm, buf, dhm.len ) ) != 0 )
240  {
241  mbedtls_printf( " failed\n ! mbedtls_dhm_read_public returned %d\n\n", ret );
242  goto exit;
243  }
244 
245  /*
246  * 7. Derive the shared secret: K = Ys ^ Xc mod P
247  */
248  mbedtls_printf( "\n . Shared secret: " );
249  fflush( stdout );
250 
251  if( ( ret = mbedtls_dhm_calc_secret( &dhm, buf, sizeof( buf ), &n,
252  mbedtls_ctr_drbg_random, &ctr_drbg ) ) != 0 )
253  {
254  mbedtls_printf( " failed\n ! mbedtls_dhm_calc_secret returned %d\n\n", ret );
255  goto exit;
256  }
257 
258  for( n = 0; n < 16; n++ )
259  mbedtls_printf( "%02x", buf[n] );
260 
261  /*
262  * 8. Setup the AES-256 encryption key
263  *
264  * This is an overly simplified example; best practice is
265  * to hash the shared secret with a random value to derive
266  * the keying material for the encryption/decryption keys
267  * and MACs.
268  */
269  mbedtls_printf( "...\n . Encrypting and sending the ciphertext" );
270  fflush( stdout );
271 
272  mbedtls_aes_setkey_enc( &aes, buf, 256 );
273  memcpy( buf, PLAINTEXT, 16 );
274  mbedtls_aes_crypt_ecb( &aes, MBEDTLS_AES_ENCRYPT, buf, buf );
275 
276  if( ( ret = mbedtls_net_send( &client_fd, buf, 16 ) ) != 16 )
277  {
278  mbedtls_printf( " failed\n ! mbedtls_net_send returned %d\n\n", ret );
279  goto exit;
280  }
281 
282  mbedtls_printf( "\n\n" );
283 
284 exit:
285 
286  mbedtls_net_free( &client_fd );
287  mbedtls_net_free( &listen_fd );
288 
289  mbedtls_aes_free( &aes );
290  mbedtls_rsa_free( &rsa );
291  mbedtls_dhm_free( &dhm );
292  mbedtls_ctr_drbg_free( &ctr_drbg );
293  mbedtls_entropy_free( &entropy );
294 
295 #if defined(_WIN32)
296  mbedtls_printf( " + Press Enter to exit this program.\n" );
297  fflush( stdout ); getchar();
298 #endif
299 
300  return( ret );
301 }
