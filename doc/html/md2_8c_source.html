109 void mbedtls_md2_starts( mbedtls_md2_context *ctx )
110 {
111  memset( ctx->cksum, 0, 16 );
112  memset( ctx->state, 0, 46 );
113  memset( ctx->buffer, 0, 16 );
114  ctx->left = 0;
115 }

117 #if !defined(MBEDTLS_MD2_PROCESS_ALT)
118 void mbedtls_md2_process( mbedtls_md2_context *ctx )
119 {
120  int i, j;
121  unsigned char t = 0;
122 
123  for( i = 0; i < 16; i++ )
124  {
125  ctx->state[i + 16] = ctx->buffer[i];
126  ctx->state[i + 32] =
127  (unsigned char)( ctx->buffer[i] ^ ctx->state[i]);
128  }
129 
130  for( i = 0; i < 18; i++ )
131  {
132  for( j = 0; j < 48; j++ )
133  {
134  ctx->state[j] = (unsigned char)
135  ( ctx->state[j] ^ PI_SUBST[t] );
136  t = ctx->state[j];
137  }
138 
139  t = (unsigned char)( t + i );
140  }
141 
142  t = ctx->cksum[15];
143 
144  for( i = 0; i < 16; i++ )
145  {
146  ctx->cksum[i] = (unsigned char)
147  ( ctx->cksum[i] ^ PI_SUBST[ctx->buffer[i] ^ t] );
148  t = ctx->cksum[i];
149  }
150 }
151 #endif /* !MBEDTLS_MD2_PROCESS_ALT */

153 /*
154  * MD2 process buffer
155  */
156 void mbedtls_md2_update( mbedtls_md2_context *ctx, const unsigned char *input, size_t ilen )
157 {
158  size_t fill;
159 
160  while( ilen > 0 )
161  {
162  if( ctx->left + ilen > 16 )
163  fill = 16 - ctx->left;
164  else
165  fill = ilen;
166 
167  memcpy( ctx->buffer + ctx->left, input, fill );
168 
169  ctx->left += fill;
170  input += fill;
171  ilen -= fill;
172 
173  if( ctx->left == 16 )
174  {
175  ctx->left = 0;
176  mbedtls_md2_process( ctx );
177  }
178  }
179 }

181 /*
182  * MD2 final digest
183  */
184 void mbedtls_md2_finish( mbedtls_md2_context *ctx, unsigned char output[16] )
185 {
186  size_t i;
187  unsigned char x;
188 
189  x = (unsigned char)( 16 - ctx->left );
190 
191  for( i = ctx->left; i < 16; i++ )
192  ctx->buffer[i] = x;
193 
194  mbedtls_md2_process( ctx );
195 
196  memcpy( ctx->buffer, ctx->cksum, 16 );
197  mbedtls_md2_process( ctx );
198 
199  memcpy( output, ctx->state, 16 );
200 }

204 /*
205  * output = MD2( input buffer )
206  */
207 void mbedtls_md2( const unsigned char *input, size_t ilen, unsigned char output[16] )
208 {
209  mbedtls_md2_context ctx;
210 
211  mbedtls_md2_init( &ctx );
212  mbedtls_md2_starts( &ctx );
213  mbedtls_md2_update( &ctx, input, ilen );
214  mbedtls_md2_finish( &ctx, output );
215  mbedtls_md2_free( &ctx );
216 }
