72 static int rsa_verify_wrap( void *ctx, mbedtls_md_type_t md_alg,
73  const unsigned char *hash, size_t hash_len,
74  const unsigned char *sig, size_t sig_len )
75 {
76  int ret;
77 
78  if( sig_len < ((mbedtls_rsa_context *) ctx)->len )
79  return( MBEDTLS_ERR_RSA_VERIFY_FAILED );
80 
81  if( ( ret = mbedtls_rsa_pkcs1_verify( (mbedtls_rsa_context *) ctx, NULL, NULL,
82  MBEDTLS_RSA_PUBLIC, md_alg,
83  (unsigned int) hash_len, hash, sig ) ) != 0 )
84  return( ret );
85 
86  if( sig_len > ((mbedtls_rsa_context *) ctx)->len )
87  return( MBEDTLS_ERR_PK_SIG_LEN_MISMATCH );
88 
89  return( 0 );
90 }

429 #if defined(MBEDTLS_RSA_C)
430 static int rsa_alt_check_pair( const void *pub, const void *prv )
431 {
432  unsigned char sig[MBEDTLS_MPI_MAX_SIZE];
433  unsigned char hash[32];
434  size_t sig_len = 0;
435  int ret;
436 
437  if( rsa_alt_get_bitlen( prv ) != rsa_get_bitlen( pub ) )
438  return( MBEDTLS_ERR_RSA_KEY_CHECK_FAILED );
439 
440  memset( hash, 0x2a, sizeof( hash ) );
441 
442  if( ( ret = rsa_alt_sign_wrap( (void *) prv, MBEDTLS_MD_NONE,
443  hash, sizeof( hash ),
444  sig, &sig_len, NULL, NULL ) ) != 0 )
445  {
446  return( ret );
447  }
448 
449  if( rsa_verify_wrap( (void *) pub, MBEDTLS_MD_NONE,
450  hash, sizeof( hash ), sig, sig_len ) != 0 )
451  {
452  return( MBEDTLS_ERR_RSA_KEY_CHECK_FAILED );
453  }
454 
455  return( 0 );
456 }
457 #endif /* MBEDTLS_RSA_C */
