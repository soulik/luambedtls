465 static inline int mbedtls_ssl_safer_memcmp( const void *a, const void *b, size_t n )
466 {
467  size_t i;
468  const unsigned char *A = (const unsigned char *) a;
469  const unsigned char *B = (const unsigned char *) b;
470  unsigned char diff = 0;
471 
472  for( i = 0; i < n; i++ )
