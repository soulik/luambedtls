55 static int pkcs5_parse_pbkdf2_params( const mbedtls_asn1_buf *params,
56  mbedtls_asn1_buf *salt, int *iterations,
57  int *keylen, mbedtls_md_type_t *md_type )
58 {
59  int ret;
60  mbedtls_asn1_buf prf_alg_oid;
61  unsigned char *p = params->p;
62  const unsigned char *end = params->p + params->len;
63 
64  if( params->tag != ( MBEDTLS_ASN1_CONSTRUCTED | MBEDTLS_ASN1_SEQUENCE ) )
65  return( MBEDTLS_ERR_PKCS5_INVALID_FORMAT +
66  MBEDTLS_ERR_ASN1_UNEXPECTED_TAG );
67  /*
68  * PBKDF2-params ::= SEQUENCE {
69  * salt OCTET STRING,
70  * iterationCount INTEGER,
71  * keyLength INTEGER OPTIONAL
72  * prf AlgorithmIdentifier DEFAULT algid-hmacWithSHA1
73  * }
74  *
75  */
76  if( ( ret = mbedtls_asn1_get_tag( &p, end, &salt->len, MBEDTLS_ASN1_OCTET_STRING ) ) != 0 )
77  return( MBEDTLS_ERR_PKCS5_INVALID_FORMAT + ret );
78 
79  salt->p = p;
80  p += salt->len;
81 
82  if( ( ret = mbedtls_asn1_get_int( &p, end, iterations ) ) != 0 )
83  return( MBEDTLS_ERR_PKCS5_INVALID_FORMAT + ret );
84 
85  if( p == end )
86  return( 0 );
87 
88  if( ( ret = mbedtls_asn1_get_int( &p, end, keylen ) ) != 0 )
89  {
90  if( ret != MBEDTLS_ERR_ASN1_UNEXPECTED_TAG )
91  return( MBEDTLS_ERR_PKCS5_INVALID_FORMAT + ret );
92  }
93 
94  if( p == end )
95  return( 0 );
96 
97  if( ( ret = mbedtls_asn1_get_alg_null( &p, end, &prf_alg_oid ) ) != 0 )
98  return( MBEDTLS_ERR_PKCS5_INVALID_FORMAT + ret );
99 
100  if( MBEDTLS_OID_CMP( MBEDTLS_OID_HMAC_SHA1, &prf_alg_oid ) != 0 )
101  return( MBEDTLS_ERR_PKCS5_FEATURE_UNAVAILABLE );
102 
103  *md_type = MBEDTLS_MD_SHA1;
104 
105  if( p != end )
106  return( MBEDTLS_ERR_PKCS5_INVALID_FORMAT +
107  MBEDTLS_ERR_ASN1_LENGTH_MISMATCH );
108 
109  return( 0 );
110 }

191

193  goto exit;
194 
195  if( ( ret = mbedtls_pkcs5_pbkdf2_hmac( &md_ctx, pwd, pwdlen, salt.p, salt.len,
196  iterations, keylen, key ) ) != 0 )
197  {
198  goto exit;

202  goto exit;
203

205  goto exit;

208  data, datalen, output, &olen ) ) != 0 )
209  ret = MBEDTLS_ERR_PKCS5_PASSWORD_MISMATCH;

211 exit:

213  mbedtls_cipher_free( &cipher_ctx );

218 int mbedtls_pkcs5_pbkdf2_hmac( mbedtls_md_context_t *ctx, const unsigned char *password,
219  size_t plen, const unsigned char *salt, size_t slen,
220  unsigned int iteration_count,
221  uint32_t key_length, unsigned char *output )
222 {
223  int ret, j;
224  unsigned int i;
225  unsigned char md1[MBEDTLS_MD_MAX_SIZE];
226  unsigned char work[MBEDTLS_MD_MAX_SIZE];
227  unsigned char md_size = mbedtls_md_get_size( ctx->md_info );
228  size_t use_len;
229  unsigned char *out_p = output;
230  unsigned char counter[4];
231 
232  memset( counter, 0, 4 );
233  counter[3] = 1;
234 
235  if( iteration_count > 0xFFFFFFFF )
236  return( MBEDTLS_ERR_PKCS5_BAD_INPUT_DATA );
237 
238  while( key_length )
239  {
240  // U1 ends up in work
241  //
242  if( ( ret = mbedtls_md_hmac_starts( ctx, password, plen ) ) != 0 )
243  return( ret );
244 
245  if( ( ret = mbedtls_md_hmac_update( ctx, salt, slen ) ) != 0 )
246  return( ret );
247 
248  if( ( ret = mbedtls_md_hmac_update( ctx, counter, 4 ) ) != 0 )
249  return( ret );
250 
251  if( ( ret = mbedtls_md_hmac_finish( ctx, work ) ) != 0 )
252  return( ret );
253 
254  memcpy( md1, work, md_size );
255 
256  for( i = 1; i < iteration_count; i++ )
257  {
258  // U2 ends up in md1
259  //
260  if( ( ret = mbedtls_md_hmac_starts( ctx, password, plen ) ) != 0 )
261  return( ret );
262 
263  if( ( ret = mbedtls_md_hmac_update( ctx, md1, md_size ) ) != 0 )
264  return( ret );
265 
266  if( ( ret = mbedtls_md_hmac_finish( ctx, md1 ) ) != 0 )
267  return( ret );
268 
269  // U1 xor U2
270  //
271  for( j = 0; j < md_size; j++ )
272  work[j] ^= md1[j];
273  }
274 
275  use_len = ( key_length < md_size ) ? key_length : md_size;
276  memcpy( out_p, work, use_len );
277 
278  key_length -= (uint32_t) use_len;
279  out_p += use_len;
280 
281  for( i = 4; i > 0; i-- )
282  if( ++counter[i - 1] != 0 )
283  break;
284  }
285 
286  return( 0 );
287 }
