52 /*
53  * Compute shared secret (SEC1 3.3.1)
54  */
55 int mbedtls_ecdh_compute_shared( mbedtls_ecp_group *grp, mbedtls_mpi *z,
56  const mbedtls_ecp_point *Q, const mbedtls_mpi *d,
57  int (*f_rng)(void *, unsigned char *, size_t),
58  void *p_rng )
59 {
60  int ret;
61  mbedtls_ecp_point P;
62 
63  mbedtls_ecp_point_init( &P );
64 
65  /*
66  * Make sure Q is a valid pubkey before using it
67  */
68  MBEDTLS_MPI_CHK( mbedtls_ecp_check_pubkey( grp, Q ) );
69 
70  MBEDTLS_MPI_CHK( mbedtls_ecp_mul( grp, &P, d, Q, f_rng, p_rng ) );
71 
72  if( mbedtls_ecp_is_zero( &P ) )
73  {
74  ret = MBEDTLS_ERR_ECP_BAD_INPUT_DATA;
75  goto cleanup;
76  }
77 
78  MBEDTLS_MPI_CHK( mbedtls_mpi_copy( z, &P.X ) );
79 
80 cleanup:
81  mbedtls_ecp_point_free( &P );
82 
83  return( ret );
84 }

112 /*
113  * Setup and write the ServerKeyExhange parameters (RFC 4492)
114  * struct {
115  * ECParameters curve_params;
116  * ECPoint public;
117  * } ServerECDHParams;
118  */
119 int mbedtls_ecdh_make_params( mbedtls_ecdh_context *ctx, size_t *olen,
120  unsigned char *buf, size_t blen,
121  int (*f_rng)(void *, unsigned char *, size_t),
122  void *p_rng )
123 {
124  int ret;
125  size_t grp_len, pt_len;
126 
127  if( ctx == NULL || ctx->grp.pbits == 0 )
128  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
129 
130  if( ( ret = mbedtls_ecdh_gen_public( &ctx->grp, &ctx->d, &ctx->Q, f_rng, p_rng ) )
131  != 0 )
132  return( ret );
133 
134  if( ( ret = mbedtls_ecp_tls_write_group( &ctx->grp, &grp_len, buf, blen ) )
135  != 0 )
136  return( ret );
137 
138  buf += grp_len;
139  blen -= grp_len;
140 
141  if( ( ret = mbedtls_ecp_tls_write_point( &ctx->grp, &ctx->Q, ctx->point_format,
142  &pt_len, buf, blen ) ) != 0 )
143  return( ret );
144 
145  *olen = grp_len + pt_len;
146  return( 0 );
147 }

149 /*
150  * Read the ServerKeyExhange parameters (RFC 4492)
151  * struct {
152  * ECParameters curve_params;
153  * ECPoint public;
154  * } ServerECDHParams;
155  */
156 int mbedtls_ecdh_read_params( mbedtls_ecdh_context *ctx,
157  const unsigned char **buf, const unsigned char *end )
158 {
159  int ret;
160 
161  if( ( ret = mbedtls_ecp_tls_read_group( &ctx->grp, buf, end - *buf ) ) != 0 )
162  return( ret );
163 
164  if( ( ret = mbedtls_ecp_tls_read_point( &ctx->grp, &ctx->Qp, buf, end - *buf ) )
165  != 0 )
166  return( ret );
167 
168  return( 0 );
169 }

171 /*
172  * Get parameters from a keypair
173  */
174 int mbedtls_ecdh_get_params( mbedtls_ecdh_context *ctx, const mbedtls_ecp_keypair *key,
175  mbedtls_ecdh_side side )
176 {
177  int ret;
178 
179  if( ( ret = mbedtls_ecp_group_copy( &ctx->grp, &key->grp ) ) != 0 )
180  return( ret );
181 
182  /* If it's not our key, just import the public part as Qp */
183  if( side == MBEDTLS_ECDH_THEIRS )
184  return( mbedtls_ecp_copy( &ctx->Qp, &key->Q ) );
185 
186  /* Our key: import public (as Q) and private parts */
187  if( side != MBEDTLS_ECDH_OURS )
188  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
189 
190  if( ( ret = mbedtls_ecp_copy( &ctx->Q, &key->Q ) ) != 0 ||
191  ( ret = mbedtls_mpi_copy( &ctx->d, &key->d ) ) != 0 )
192  return( ret );
193 
194  return( 0 );
195 }

218 /*
219  * Parse and import the client's public value
220  */
221 int mbedtls_ecdh_read_public( mbedtls_ecdh_context *ctx,
222  const unsigned char *buf, size_t blen )
223 {
224  int ret;
225  const unsigned char *p = buf;
226 
227  if( ctx == NULL )
228  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
229 
230  if( ( ret = mbedtls_ecp_tls_read_point( &ctx->grp, &ctx->Qp, &p, blen ) ) != 0 )
231  return( ret );

234  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
235 
236  return( 0 );

239 /*
240  * Derive and export the shared secret
241  */
242 int mbedtls_ecdh_calc_secret( mbedtls_ecdh_context *ctx, size_t *olen,
243  unsigned char *buf, size_t blen,
244  int (*f_rng)(void *, unsigned char *, size_t),
245  void *p_rng )
246 {
247  int ret;
248 
249  if( ctx == NULL )
250  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
251 
252  if( ( ret = mbedtls_ecdh_compute_shared( &ctx->grp, &ctx->z, &ctx->Qp, &ctx->d,
253  f_rng, p_rng ) ) != 0 )
254  {
255  return( ret );
256  }
257 
258  if( mbedtls_mpi_size( &ctx->z ) > blen )
259  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
260 
261  *olen = ctx->grp.pbits / 8 + ( ( ctx->grp.pbits % 8 ) != 0 );
262  return mbedtls_mpi_write_binary( &ctx->z, buf, *olen );
263 }
