46 /*
47  * Derive a suitable integer for group grp from a buffer of length len
48  * SEC1 4.1.3 step 5 aka SEC1 4.1.4 step 3
49  */
50 static int derive_mpi( const mbedtls_ecp_group *grp, mbedtls_mpi *x,
51  const unsigned char *buf, size_t blen )
52 {
53  int ret;
54  size_t n_size = ( grp->nbits + 7 ) / 8;
55  size_t use_size = blen > n_size ? n_size : blen;
56 
57  MBEDTLS_MPI_CHK( mbedtls_mpi_read_binary( x, buf, use_size ) );
58  if( use_size * 8 > grp->nbits )
59  MBEDTLS_MPI_CHK( mbedtls_mpi_shift_r( x, use_size * 8 - grp->nbits ) );
60 
61  /* While at it, reduce modulo N */
62  if( mbedtls_mpi_cmp_mpi( x, &grp->N ) >= 0 )
63  MBEDTLS_MPI_CHK( mbedtls_mpi_sub_mpi( x, x, &grp->N ) );
64 
65 cleanup:
66  return( ret );
67 }

69 /*
70  * Compute ECDSA signature of a hashed message (SEC1 4.1.3)
71  * Obviously, compared to SEC1 4.1.3, we skip step 4 (hash message)
72  */
73 int mbedtls_ecdsa_sign( mbedtls_ecp_group *grp, mbedtls_mpi *r, mbedtls_mpi *s,
74  const mbedtls_mpi *d, const unsigned char *buf, size_t blen,
75  int (*f_rng)(void *, unsigned char *, size_t), void *p_rng )
76 {
77  int ret, key_tries, sign_tries, blind_tries;
78  mbedtls_ecp_point R;
79  mbedtls_mpi k, e, t;
80 
81  /* Fail cleanly on curves such as Curve25519 that can't be used for ECDSA */
82  if( grp->N.p == NULL )
83  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
84 
85  mbedtls_ecp_point_init( &R );
86  mbedtls_mpi_init( &k ); mbedtls_mpi_init( &e ); mbedtls_mpi_init( &t );
87 
88  sign_tries = 0;
89  do
90  {
91  /*
92  * Steps 1-3: generate a suitable ephemeral keypair
93  * and set r = xR mod n
94  */
95  key_tries = 0;
96  do
97  {
98  MBEDTLS_MPI_CHK( mbedtls_ecp_gen_keypair( grp, &k, &R, f_rng, p_rng ) );
99  MBEDTLS_MPI_CHK( mbedtls_mpi_mod_mpi( r, &R.X, &grp->N ) );
100 
101  if( key_tries++ > 10 )
102  {
103  ret = MBEDTLS_ERR_ECP_RANDOM_FAILED;
104  goto cleanup;
105  }
106  }
107  while( mbedtls_mpi_cmp_int( r, 0 ) == 0 );
108 
109  /*
110  * Step 5: derive MPI from hashed message
111  */
112  MBEDTLS_MPI_CHK( derive_mpi( grp, &e, buf, blen ) );
113 
114  /*
115  * Generate a random value to blind inv_mod in next step,
116  * avoiding a potential timing leak.
117  */
118  blind_tries = 0;
119  do
120  {
121  size_t n_size = ( grp->nbits + 7 ) / 8;
122  MBEDTLS_MPI_CHK( mbedtls_mpi_fill_random( &t, n_size, f_rng, p_rng ) );
123  MBEDTLS_MPI_CHK( mbedtls_mpi_shift_r( &t, 8 * n_size - grp->nbits ) );
124 
125  /* See mbedtls_ecp_gen_keypair() */
126  if( ++blind_tries > 30 )
127  return( MBEDTLS_ERR_ECP_RANDOM_FAILED );
128  }
129  while( mbedtls_mpi_cmp_int( &t, 1 ) < 0 ||
130  mbedtls_mpi_cmp_mpi( &t, &grp->N ) >= 0 );
131 
132  /*
133  * Step 6: compute s = (e + r * d) / k = t (e + rd) / (kt) mod n
134  */
135  MBEDTLS_MPI_CHK( mbedtls_mpi_mul_mpi( s, r, d ) );
136  MBEDTLS_MPI_CHK( mbedtls_mpi_add_mpi( &e, &e, s ) );
137  MBEDTLS_MPI_CHK( mbedtls_mpi_mul_mpi( &e, &e, &t ) );
138  MBEDTLS_MPI_CHK( mbedtls_mpi_mul_mpi( &k, &k, &t ) );
139  MBEDTLS_MPI_CHK( mbedtls_mpi_inv_mod( s, &k, &grp->N ) );
140  MBEDTLS_MPI_CHK( mbedtls_mpi_mul_mpi( s, s, &e ) );
141  MBEDTLS_MPI_CHK( mbedtls_mpi_mod_mpi( s, s, &grp->N ) );
142 
143  if( sign_tries++ > 10 )
144  {
145  ret = MBEDTLS_ERR_ECP_RANDOM_FAILED;
146  goto cleanup;
147  }
148  }
149  while( mbedtls_mpi_cmp_int( s, 0 ) == 0 );
150 
151 cleanup:
152  mbedtls_ecp_point_free( &R );
153  mbedtls_mpi_free( &k ); mbedtls_mpi_free( &e ); mbedtls_mpi_free( &t );
154 
155  return( ret );
156 }

159 /*
160  * Deterministic signature wrapper
161  */
162 int mbedtls_ecdsa_sign_det( mbedtls_ecp_group *grp, mbedtls_mpi *r, mbedtls_mpi *s,
163  const mbedtls_mpi *d, const unsigned char *buf, size_t blen,
164  mbedtls_md_type_t md_alg )
165 {
166  int ret;
167  mbedtls_hmac_drbg_context rng_ctx;
168  unsigned char data[2 * MBEDTLS_ECP_MAX_BYTES];
169  size_t grp_len = ( grp->nbits + 7 ) / 8;
170  const mbedtls_md_info_t *md_info;
171  mbedtls_mpi h;
172 
173  if( ( md_info = mbedtls_md_info_from_type( md_alg ) ) == NULL )
174  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
175 
176  mbedtls_mpi_init( &h );
177  mbedtls_hmac_drbg_init( &rng_ctx );
178 
179  /* Use private key and message hash (reduced) to initialize HMAC_DRBG */
180  MBEDTLS_MPI_CHK( mbedtls_mpi_write_binary( d, data, grp_len ) );
181  MBEDTLS_MPI_CHK( derive_mpi( grp, &h, buf, blen ) );
182  MBEDTLS_MPI_CHK( mbedtls_mpi_write_binary( &h, data + grp_len, grp_len ) );
183  mbedtls_hmac_drbg_seed_buf( &rng_ctx, md_info, data, 2 * grp_len );
184 
185  ret = mbedtls_ecdsa_sign( grp, r, s, d, buf, blen,
186  mbedtls_hmac_drbg_random, &rng_ctx );
187 
188 cleanup:
189  mbedtls_hmac_drbg_free( &rng_ctx );
190  mbedtls_mpi_free( &h );
191 
192  return( ret );
193 }

196 /*
197  * Verify ECDSA signature of hashed message (SEC1 4.1.4)
198  * Obviously, compared to SEC1 4.1.3, we skip step 2 (hash message)
199  */
200 int mbedtls_ecdsa_verify( mbedtls_ecp_group *grp,
201  const unsigned char *buf, size_t blen,
202  const mbedtls_ecp_point *Q, const mbedtls_mpi *r, const mbedtls_mpi *s)
203 {
204  int ret;
205  mbedtls_mpi e, s_inv, u1, u2;
206  mbedtls_ecp_point R;
207 
208  mbedtls_ecp_point_init( &R );
209  mbedtls_mpi_init( &e ); mbedtls_mpi_init( &s_inv ); mbedtls_mpi_init( &u1 ); mbedtls_mpi_init( &u2 );
210 
211  /* Fail cleanly on curves such as Curve25519 that can't be used for ECDSA */
212  if( grp->N.p == NULL )
213  return( MBEDTLS_ERR_ECP_BAD_INPUT_DATA );
214 
215  /*
216  * Step 1: make sure r and s are in range 1..n-1
217  */
218  if( mbedtls_mpi_cmp_int( r, 1 ) < 0 || mbedtls_mpi_cmp_mpi( r, &grp->N ) >= 0 ||
219  mbedtls_mpi_cmp_int( s, 1 ) < 0 || mbedtls_mpi_cmp_mpi( s, &grp->N ) >= 0 )
220  {
221  ret = MBEDTLS_ERR_ECP_VERIFY_FAILED;
222  goto cleanup;
223  }
224 
225  /*
226  * Additional precaution: make sure Q is valid
227  */
228  MBEDTLS_MPI_CHK( mbedtls_ecp_check_pubkey( grp, Q ) );
229 
230  /*
231  * Step 3: derive MPI from hashed message
232  */
233  MBEDTLS_MPI_CHK( derive_mpi( grp, &e, buf, blen ) );
234 
235  /*
236  * Step 4: u1 = e / s mod n, u2 = r / s mod n
237  */
238  MBEDTLS_MPI_CHK( mbedtls_mpi_inv_mod( &s_inv, s, &grp->N ) );
239 
240  MBEDTLS_MPI_CHK( mbedtls_mpi_mul_mpi( &u1, &e, &s_inv ) );
241  MBEDTLS_MPI_CHK( mbedtls_mpi_mod_mpi( &u1, &u1, &grp->N ) );
242 
243  MBEDTLS_MPI_CHK( mbedtls_mpi_mul_mpi( &u2, r, &s_inv ) );
244  MBEDTLS_MPI_CHK( mbedtls_mpi_mod_mpi( &u2, &u2, &grp->N ) );
245 
246  /*
247  * Step 5: R = u1 G + u2 Q
248  *
249  * Since we're not using any secret data, no need to pass a RNG to
250  * mbedtls_ecp_mul() for countermesures.
251  */
252  MBEDTLS_MPI_CHK( mbedtls_ecp_muladd( grp, &R, &u1, &grp->G, &u2, Q ) );
253 
254  if( mbedtls_ecp_is_zero( &R ) )
255  {
256  ret = MBEDTLS_ERR_ECP_VERIFY_FAILED;
257  goto cleanup;
258  }
259 
260  /*
261  * Step 6: convert xR to an integer (no-op)
262  * Step 7: reduce xR mod n (gives v)
263  */
264  MBEDTLS_MPI_CHK( mbedtls_mpi_mod_mpi( &R.X, &R.X, &grp->N ) );
265 
266  /*
267  * Step 8: check if v (that is, R.X) is equal to r
268  */
269  if( mbedtls_mpi_cmp_mpi( &R.X, r ) != 0 )
270  {
271  ret = MBEDTLS_ERR_ECP_VERIFY_FAILED;
272  goto cleanup;
273  }
274 
275 cleanup:
276  mbedtls_ecp_point_free( &R );
277  mbedtls_mpi_free( &e ); mbedtls_mpi_free( &s_inv ); mbedtls_mpi_free( &u1 ); mbedtls_mpi_free( &u2 );
278 
279  return( ret );
280 }

282 /*
283  * Convert a signature (given by context) to ASN.1
284  */
285 static int ecdsa_signature_to_asn1( const mbedtls_mpi *r, const mbedtls_mpi *s,
286  unsigned char *sig, size_t *slen )
287 {
288  int ret;
289  unsigned char buf[MBEDTLS_ECDSA_MAX_LEN];
290  unsigned char *p = buf + sizeof( buf );
291  size_t len = 0;
292 
293  MBEDTLS_ASN1_CHK_ADD( len, mbedtls_asn1_write_mpi( &p, buf, s ) );
294  MBEDTLS_ASN1_CHK_ADD( len, mbedtls_asn1_write_mpi( &p, buf, r ) );
295 
296  MBEDTLS_ASN1_CHK_ADD( len, mbedtls_asn1_write_len( &p, buf, len ) );
297  MBEDTLS_ASN1_CHK_ADD( len, mbedtls_asn1_write_tag( &p, buf,
298  MBEDTLS_ASN1_CONSTRUCTED | MBEDTLS_ASN1_SEQUENCE ) );
299 
300  memcpy( sig, p, len );
301  *slen = len;
302 
303  return( 0 );
304 }

310  const unsigned char *hash, size_t hlen,

324 
325  MBEDTLS_MPI_CHK( mbedtls_ecdsa_sign_det( &ctx->grp, &r, &s, &ctx->d,

336 cleanup:

338  mbedtls_mpi_free( &s );
339

355 /*
356  * Read and check signature
357  */
358 int mbedtls_ecdsa_read_signature( mbedtls_ecdsa_context *ctx,
359  const unsigned char *hash, size_t hlen,
360  const unsigned char *sig, size_t slen )
361 {
362  int ret;
363  unsigned char *p = (unsigned char *) sig;
364  const unsigned char *end = sig + slen;
365  size_t len;
366  mbedtls_mpi r, s;
367 
368  mbedtls_mpi_init( &r );
369  mbedtls_mpi_init( &s );
370 
371  if( ( ret = mbedtls_asn1_get_tag( &p, end, &len,
372  MBEDTLS_ASN1_CONSTRUCTED | MBEDTLS_ASN1_SEQUENCE ) ) != 0 )
373  {
374  ret += MBEDTLS_ERR_ECP_BAD_INPUT_DATA;
375  goto cleanup;
376  }
377 
378  if( p + len != end )
379  {
380  ret = MBEDTLS_ERR_ECP_BAD_INPUT_DATA +
381  MBEDTLS_ERR_ASN1_LENGTH_MISMATCH;
382  goto cleanup;
383  }
384 
385  if( ( ret = mbedtls_asn1_get_mpi( &p, end, &r ) ) != 0 ||
386  ( ret = mbedtls_asn1_get_mpi( &p, end, &s ) ) != 0 )
387  {
388  ret += MBEDTLS_ERR_ECP_BAD_INPUT_DATA;
389  goto cleanup;
390  }
391 
392  if( ( ret = mbedtls_ecdsa_verify( &ctx->grp, hash, hlen,
393  &ctx->Q, &r, &s ) ) != 0 )
394  goto cleanup;
395 
396  if( p != end )
397  ret = MBEDTLS_ERR_ECP_SIG_LEN_MISMATCH;
398 
399 cleanup:
400  mbedtls_mpi_free( &r );
401  mbedtls_mpi_free( &s );
402 
403  return( ret );
404 }
