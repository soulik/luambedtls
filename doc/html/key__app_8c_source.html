75  * global options
76  */
77 struct options
78 {
79  int mode; /* the mode to run the application in */
80  const char *filename; /* filename of the key file */
81  const char *password; /* password for the private key */
82  const char *password_file; /* password_file for the private key */
83 } opt;
84 
85 int main( int argc, char *argv[] )
86 {
87  int ret = 0;
88  mbedtls_pk_context pk;
89  char buf[1024];
90  int i;
91  char *p, *q;
92 
93  /*
94  * Set to sane values
95  */
96  mbedtls_pk_init( &pk );
97  memset( buf, 0, sizeof(buf) );
98 
99  if( argc == 0 )
100  {
101  usage:
102  mbedtls_printf( USAGE );
103  goto exit;
104  }
105 
106  opt.mode = DFL_MODE;
107  opt.filename = DFL_FILENAME;
108  opt.password = DFL_PASSWORD;
109  opt.password_file = DFL_PASSWORD_FILE;
110 
111  for( i = 1; i < argc; i++ )
112  {
113  p = argv[i];
114  if( ( q = strchr( p, '=' ) ) == NULL )
115  goto usage;
116  *q++ = '\0';
117 
118  if( strcmp( p, "mode" ) == 0 )
119  {
120  if( strcmp( q, "private" ) == 0 )
121  opt.mode = MODE_PRIVATE;
122  else if( strcmp( q, "public" ) == 0 )
123  opt.mode = MODE_PUBLIC;
124  else
125  goto usage;
126  }
127  else if( strcmp( p, "filename" ) == 0 )
128  opt.filename = q;
129  else if( strcmp( p, "password" ) == 0 )
130  opt.password = q;
131  else if( strcmp( p, "password_file" ) == 0 )
132  opt.password_file = q;
133  else
134  goto usage;
135  }
136 
137  if( opt.mode == MODE_PRIVATE )
138  {
139  if( strlen( opt.password ) && strlen( opt.password_file ) )
140  {
141  mbedtls_printf( "Error: cannot have both password and password_file\n" );
142  goto usage;
143  }
144 
145  if( strlen( opt.password_file ) )
146  {
147  FILE *f;
148 
149  mbedtls_printf( "\n . Loading the password file ..." );
150  if( ( f = fopen( opt.password_file, "rb" ) ) == NULL )
151  {
152  mbedtls_printf( " failed\n ! fopen returned NULL\n" );
153  goto exit;
154  }
155  if( fgets( buf, sizeof(buf), f ) == NULL )
156  {
157  fclose( f );
158  mbedtls_printf( "Error: fgets() failed to retrieve password\n" );
159  goto exit;
160  }
161  fclose( f );
162 
163  i = (int) strlen( buf );
164  if( buf[i - 1] == '\n' ) buf[i - 1] = '\0';
165  if( buf[i - 2] == '\r' ) buf[i - 2] = '\0';
166  opt.password = buf;
167  }
168 
169  /*
170  * 1.1. Load the key
171  */
172  mbedtls_printf( "\n . Loading the private key ..." );
173  fflush( stdout );
174 
175  ret = mbedtls_pk_parse_keyfile( &pk, opt.filename, opt.password );
176 
177  if( ret != 0 )
178  {
179  mbedtls_printf( " failed\n ! mbedtls_pk_parse_keyfile returned -0x%04x\n", -ret );
180  goto exit;
181  }
182 
183  mbedtls_printf( " ok\n" );
184 
185  /*
186  * 1.2 Print the key
187  */
188  mbedtls_printf( " . Key information ...\n" );
189 #if defined(MBEDTLS_RSA_C)
190  if( mbedtls_pk_get_type( &pk ) == MBEDTLS_PK_RSA )
191  {
192  mbedtls_rsa_context *rsa = mbedtls_pk_rsa( pk );
193  mbedtls_mpi_write_file( "N: ", &rsa->N, 16, NULL );
194  mbedtls_mpi_write_file( "E: ", &rsa->E, 16, NULL );
195  mbedtls_mpi_write_file( "D: ", &rsa->D, 16, NULL );
196  mbedtls_mpi_write_file( "P: ", &rsa->P, 16, NULL );
197  mbedtls_mpi_write_file( "Q: ", &rsa->Q, 16, NULL );
198  mbedtls_mpi_write_file( "DP: ", &rsa->DP, 16, NULL );
199  mbedtls_mpi_write_file( "DQ: ", &rsa->DQ, 16, NULL );
200  mbedtls_mpi_write_file( "QP: ", &rsa->QP, 16, NULL );
201  }
202  else
203 #endif
204 #if defined(MBEDTLS_ECP_C)
205  if( mbedtls_pk_get_type( &pk ) == MBEDTLS_PK_ECKEY )
206  {
207  mbedtls_ecp_keypair *ecp = mbedtls_pk_ec( pk );
208  mbedtls_mpi_write_file( "Q(X): ", &ecp->Q.X, 16, NULL );
209  mbedtls_mpi_write_file( "Q(Y): ", &ecp->Q.Y, 16, NULL );
210  mbedtls_mpi_write_file( "Q(Z): ", &ecp->Q.Z, 16, NULL );
211  mbedtls_mpi_write_file( "D : ", &ecp->d , 16, NULL );
212  }
213  else
214 #endif
215  {
216  mbedtls_printf("Do not know how to print key information for this type\n" );
217  goto exit;
218  }
219  }
220  else if( opt.mode == MODE_PUBLIC )
221  {
222  /*
223  * 1.1. Load the key
224  */
225  mbedtls_printf( "\n . Loading the public key ..." );
226  fflush( stdout );
227 
228  ret = mbedtls_pk_parse_public_keyfile( &pk, opt.filename );
229 
230  if( ret != 0 )
231  {
232  mbedtls_printf( " failed\n ! mbedtls_pk_parse_public_keyfile returned -0x%04x\n", -ret );
233  goto exit;
234  }
235 
236  mbedtls_printf( " ok\n" );
237 
238  mbedtls_printf( " . Key information ...\n" );
239 #if defined(MBEDTLS_RSA_C)
240  if( mbedtls_pk_get_type( &pk ) == MBEDTLS_PK_RSA )
241  {
242  mbedtls_rsa_context *rsa = mbedtls_pk_rsa( pk );
243  mbedtls_mpi_write_file( "N: ", &rsa->N, 16, NULL );
244  mbedtls_mpi_write_file( "E: ", &rsa->E, 16, NULL );
245  }
246  else
247 #endif
248 #if defined(MBEDTLS_ECP_C)
249  if( mbedtls_pk_get_type( &pk ) == MBEDTLS_PK_ECKEY )
250  {
251  mbedtls_ecp_keypair *ecp = mbedtls_pk_ec( pk );
252  mbedtls_mpi_write_file( "Q(X): ", &ecp->Q.X, 16, NULL );
253  mbedtls_mpi_write_file( "Q(Y): ", &ecp->Q.Y, 16, NULL );
254  mbedtls_mpi_write_file( "Q(Z): ", &ecp->Q.Z, 16, NULL );
255  }
256  else
257 #endif
258  {
259  mbedtls_printf("Do not know how to print key information for this type\n" );
260  goto exit;
261  }
262  }
263  else
264  goto usage;
265 
266 exit:
267 
268 #if defined(MBEDTLS_ERROR_C)
269  mbedtls_strerror( ret, buf, sizeof(buf) );
270  mbedtls_printf( " ! Last error was: %s\n", buf );
271 #endif
272 
273  mbedtls_pk_free( &pk );
274 
275 #if defined(_WIN32)
276  mbedtls_printf( " + Press Enter to exit this program.\n" );
277  fflush( stdout ); getchar();
278 #endif
279 
280  return( ret );
281 }
