123 static int my_verify( void *data, mbedtls_x509_crt *crt, int depth, uint32_t *flags )
124 {
125  char buf[1024];
126  ((void) data);
127 
128  mbedtls_printf( "\nVerify requested for (Depth %d):\n", depth );
129  mbedtls_x509_crt_info( buf, sizeof( buf ) - 1, "", crt );
130  mbedtls_printf( "%s", buf );
131 
132  if ( ( *flags ) == 0 )
133  mbedtls_printf( " This certificate has no flags\n" );
134  else
135  {
136  mbedtls_x509_crt_verify_info( buf, sizeof( buf ), " ! ", *flags );
137  mbedtls_printf( "%s\n", buf );
138  }
139 
140  return( 0 );
141 }

143 int main( int argc, char *argv[] )
144 {
145  int ret = 0;
146  mbedtls_net_context server_fd;
147  unsigned char buf[1024];
148  mbedtls_entropy_context entropy;
149  mbedtls_ctr_drbg_context ctr_drbg;
150  mbedtls_ssl_context ssl;
151  mbedtls_ssl_config conf;
152  mbedtls_x509_crt cacert;
153  mbedtls_x509_crt clicert;
154  mbedtls_x509_crl cacrl;
155  mbedtls_pk_context pkey;
156  int i, j;
157  uint32_t flags;
158  int verify = 0;
159  char *p, *q;
160  const char *pers = "cert_app";
161 
162  /*
163  * Set to sane values
164  */
165  mbedtls_net_init( &server_fd );
166  mbedtls_ctr_drbg_init( &ctr_drbg );
167  mbedtls_ssl_init( &ssl );
168  mbedtls_ssl_config_init( &conf );
169  mbedtls_x509_crt_init( &cacert );
170  mbedtls_x509_crt_init( &clicert );
171 #if defined(MBEDTLS_X509_CRL_PARSE_C)
172  mbedtls_x509_crl_init( &cacrl );
173 #else
174  /* Zeroize structure as CRL parsing is not supported and we have to pass
175  it to the verify function */
176  memset( &cacrl, 0, sizeof(mbedtls_x509_crl) );
177 #endif
178  mbedtls_pk_init( &pkey );
179 
180  if( argc == 0 )
181  {
182  usage:
183  mbedtls_printf( USAGE );
184  ret = 2;
185  goto exit;
186  }
187 
188  opt.mode = DFL_MODE;
189  opt.filename = DFL_FILENAME;
190  opt.ca_file = DFL_CA_FILE;
191  opt.crl_file = DFL_CRL_FILE;
192  opt.ca_path = DFL_CA_PATH;
193  opt.server_name = DFL_SERVER_NAME;
194  opt.server_port = DFL_SERVER_PORT;
195  opt.debug_level = DFL_DEBUG_LEVEL;
196  opt.permissive = DFL_PERMISSIVE;
197 
198  for( i = 1; i < argc; i++ )
199  {
200  p = argv[i];
201  if( ( q = strchr( p, '=' ) ) == NULL )
202  goto usage;
203  *q++ = '\0';
204 
205  for( j = 0; p + j < q; j++ )
206  {
207  if( argv[i][j] >= 'A' && argv[i][j] <= 'Z' )
208  argv[i][j] |= 0x20;
209  }
210 
211  if( strcmp( p, "mode" ) == 0 )
212  {
213  if( strcmp( q, "file" ) == 0 )
214  opt.mode = MODE_FILE;
215  else if( strcmp( q, "ssl" ) == 0 )
216  opt.mode = MODE_SSL;
217  else
218  goto usage;
219  }
220  else if( strcmp( p, "filename" ) == 0 )
221  opt.filename = q;
222  else if( strcmp( p, "ca_file" ) == 0 )
223  opt.ca_file = q;
224  else if( strcmp( p, "crl_file" ) == 0 )
225  opt.crl_file = q;
226  else if( strcmp( p, "ca_path" ) == 0 )
227  opt.ca_path = q;
228  else if( strcmp( p, "server_name" ) == 0 )
229  opt.server_name = q;
230  else if( strcmp( p, "server_port" ) == 0 )
231  opt.server_port = q;
232  else if( strcmp( p, "debug_level" ) == 0 )
233  {
234  opt.debug_level = atoi( q );
235  if( opt.debug_level < 0 || opt.debug_level > 65535 )
236  goto usage;
237  }
238  else if( strcmp( p, "permissive" ) == 0 )
239  {
240  opt.permissive = atoi( q );
241  if( opt.permissive < 0 || opt.permissive > 1 )
242  goto usage;
243  }
244  else
245  goto usage;
246  }
247 
248  /*
249  * 1.1. Load the trusted CA
250  */
251  mbedtls_printf( " . Loading the CA root certificate ..." );
252  fflush( stdout );
253 
254  if( strlen( opt.ca_path ) )
255  {
256  ret = mbedtls_x509_crt_parse_path( &cacert, opt.ca_path );
257  verify = 1;
258  }
259  else if( strlen( opt.ca_file ) )
260  {
261  ret = mbedtls_x509_crt_parse_file( &cacert, opt.ca_file );
262  verify = 1;
263  }
264 
265  if( ret < 0 )
266  {
267  mbedtls_printf( " failed\n ! mbedtls_x509_crt_parse returned -0x%x\n\n", -ret );
268  goto exit;
269  }
270 
271  mbedtls_printf( " ok (%d skipped)\n", ret );
272 
273 #if defined(MBEDTLS_X509_CRL_PARSE_C)
274  if( strlen( opt.crl_file ) )
275  {
276  if( ( ret = mbedtls_x509_crl_parse_file( &cacrl, opt.crl_file ) ) != 0 )
277  {
278  mbedtls_printf( " failed\n ! mbedtls_x509_crl_parse returned -0x%x\n\n", -ret );
279  goto exit;
280  }
281 
282  verify = 1;
283  }
284 #endif
285 
286  if( opt.mode == MODE_FILE )
287  {
288  mbedtls_x509_crt crt;
289  mbedtls_x509_crt *cur = &crt;
290  mbedtls_x509_crt_init( &crt );
291 
292  /*
293  * 1.1. Load the certificate(s)
294  */
295  mbedtls_printf( "\n . Loading the certificate(s) ..." );
296  fflush( stdout );
297 
298  ret = mbedtls_x509_crt_parse_file( &crt, opt.filename );
299 
300  if( ret < 0 )
301  {
302  mbedtls_printf( " failed\n ! mbedtls_x509_crt_parse_file returned %d\n\n", ret );
303  mbedtls_x509_crt_free( &crt );
304  goto exit;
305  }
306 
307  if( opt.permissive == 0 && ret > 0 )
308  {
309  mbedtls_printf( " failed\n ! mbedtls_x509_crt_parse failed to parse %d certificates\n\n", ret );
310  mbedtls_x509_crt_free( &crt );
311  goto exit;
312  }
313 
314  mbedtls_printf( " ok\n" );
315 
316  /*
317  * 1.2 Print the certificate(s)
318  */
319  while( cur != NULL )
320  {
321  mbedtls_printf( " . Peer certificate information ...\n" );
322  ret = mbedtls_x509_crt_info( (char *) buf, sizeof( buf ) - 1, " ",
323  cur );
324  if( ret == -1 )
325  {
326  mbedtls_printf( " failed\n ! mbedtls_x509_crt_info returned %d\n\n", ret );
327  mbedtls_x509_crt_free( &crt );
328  goto exit;
329  }
330 
331  mbedtls_printf( "%s\n", buf );
332 
333  cur = cur->next;
334  }
335 
336  ret = 0;
337 
338  /*
339  * 1.3 Verify the certificate
340  */
341  if( verify )
342  {
343  mbedtls_printf( " . Verifying X.509 certificate..." );
344 
345  if( ( ret = mbedtls_x509_crt_verify( &crt, &cacert, &cacrl, NULL, &flags,
346  my_verify, NULL ) ) != 0 )
347  {
348  char vrfy_buf[512];
349 
350  mbedtls_printf( " failed\n" );
351 
352  mbedtls_x509_crt_verify_info( vrfy_buf, sizeof( vrfy_buf ), " ! ", flags );
353 
354  mbedtls_printf( "%s\n", vrfy_buf );
355  }
356  else
357  mbedtls_printf( " ok\n" );
358  }
359 
360  mbedtls_x509_crt_free( &crt );
361  }
362  else if( opt.mode == MODE_SSL )
363  {
364  /*
365  * 1. Initialize the RNG and the session data
366  */
367  mbedtls_printf( "\n . Seeding the random number generator..." );
368  fflush( stdout );
369 
370  mbedtls_entropy_init( &entropy );
371  if( ( ret = mbedtls_ctr_drbg_seed( &ctr_drbg, mbedtls_entropy_func, &entropy,
372  (const unsigned char *) pers,
373  strlen( pers ) ) ) != 0 )
374  {
375  mbedtls_printf( " failed\n ! mbedtls_ctr_drbg_seed returned %d\n", ret );
376  goto ssl_exit;
377  }
378 
379  mbedtls_printf( " ok\n" );
380 
381  /*
382  * 2. Start the connection
383  */
384  mbedtls_printf( " . SSL connection to tcp/%s/%s...", opt.server_name,
385  opt.server_port );
386  fflush( stdout );
387 
388  if( ( ret = mbedtls_net_connect( &server_fd, opt.server_name,
389  opt.server_port, MBEDTLS_NET_PROTO_TCP ) ) != 0 )
390  {
391  mbedtls_printf( " failed\n ! mbedtls_net_connect returned %d\n\n", ret );
392  goto ssl_exit;
393  }
394 
395  /*
396  * 3. Setup stuff
397  */
398  if( ( ret = mbedtls_ssl_config_defaults( &conf,
399  MBEDTLS_SSL_IS_CLIENT,
400  MBEDTLS_SSL_TRANSPORT_STREAM,
401  MBEDTLS_SSL_PRESET_DEFAULT ) ) != 0 )
402  {
403  mbedtls_printf( " failed\n ! mbedtls_ssl_config_defaults returned %d\n\n", ret );
404  goto exit;
405  }
406 
407  if( verify )
408  {
409  mbedtls_ssl_conf_authmode( &conf, MBEDTLS_SSL_VERIFY_REQUIRED );
410  mbedtls_ssl_conf_ca_chain( &conf, &cacert, NULL );
411  mbedtls_ssl_conf_verify( &conf, my_verify, NULL );
412  }
413  else
414  mbedtls_ssl_conf_authmode( &conf, MBEDTLS_SSL_VERIFY_NONE );
415 
416  mbedtls_ssl_conf_rng( &conf, mbedtls_ctr_drbg_random, &ctr_drbg );
417  mbedtls_ssl_conf_dbg( &conf, my_debug, stdout );
418 
419  if( ( ret = mbedtls_ssl_conf_own_cert( &conf, &clicert, &pkey ) ) != 0 )
420  {
421  mbedtls_printf( " failed\n ! mbedtls_ssl_conf_own_cert returned %d\n\n", ret );
422  goto ssl_exit;
423  }
424 
425  if( ( ret = mbedtls_ssl_setup( &ssl, &conf ) ) != 0 )
426  {
427  mbedtls_printf( " failed\n ! mbedtls_ssl_setup returned %d\n\n", ret );
428  goto ssl_exit;
429  }
430 
431  if( ( ret = mbedtls_ssl_set_hostname( &ssl, opt.server_name ) ) != 0 )
432  {
433  mbedtls_printf( " failed\n ! mbedtls_ssl_set_hostname returned %d\n\n", ret );
434  goto ssl_exit;
435  }
436 
437  mbedtls_ssl_set_bio( &ssl, &server_fd, mbedtls_net_send, mbedtls_net_recv, NULL );
438 
439  /*
440  * 4. Handshake
441  */
442  while( ( ret = mbedtls_ssl_handshake( &ssl ) ) != 0 )
443  {
444  if( ret != MBEDTLS_ERR_SSL_WANT_READ && ret != MBEDTLS_ERR_SSL_WANT_WRITE )
445  {
446  mbedtls_printf( " failed\n ! mbedtls_ssl_handshake returned %d\n\n", ret );
447  goto ssl_exit;
448  }
449  }
450 
451  mbedtls_printf( " ok\n" );
452 
453  /*
454  * 5. Print the certificate
455  */
456  mbedtls_printf( " . Peer certificate information ...\n" );
457  ret = mbedtls_x509_crt_info( (char *) buf, sizeof( buf ) - 1, " ",
458  ssl.session->peer_cert );
459  if( ret == -1 )
460  {
461  mbedtls_printf( " failed\n ! mbedtls_x509_crt_info returned %d\n\n", ret );
462  goto ssl_exit;
463  }
464 
465  mbedtls_printf( "%s\n", buf );
466 
467  mbedtls_ssl_close_notify( &ssl );
468 
469 ssl_exit:
470  mbedtls_ssl_free( &ssl );
471  mbedtls_ssl_config_free( &conf );
472  }
473  else
474  goto usage;
475 
476 exit:
477 
478  mbedtls_net_free( &server_fd );
479  mbedtls_x509_crt_free( &cacert );
480  mbedtls_x509_crt_free( &clicert );
481 #if defined(MBEDTLS_X509_CRL_PARSE_C)
482  mbedtls_x509_crl_free( &cacrl );
483 #endif
484  mbedtls_pk_free( &pkey );
485  mbedtls_ctr_drbg_free( &ctr_drbg );
486  mbedtls_entropy_free( &entropy );
487 
488 #if defined(_WIN32)
489  mbedtls_printf( " + Press Enter to exit this program.\n" );
490  fflush( stdout ); getchar();
491 #endif
492 
493  if( ret < 0 )
494  ret = 1;
495 
496  return( ret );
497 }
