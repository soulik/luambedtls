73 /*
74  * Get pk_info structure from type
75  */
76 const mbedtls_pk_info_t * mbedtls_pk_info_from_type( mbedtls_pk_type_t pk_type )
77 {
78  switch( pk_type ) {
79 #if defined(MBEDTLS_RSA_C)
80  case MBEDTLS_PK_RSA:
81  return( &mbedtls_rsa_info );
82 #endif
83 #if defined(MBEDTLS_ECP_C)
84  case MBEDTLS_PK_ECKEY:
85  return( &mbedtls_eckey_info );
86  case MBEDTLS_PK_ECKEY_DH:
87  return( &mbedtls_eckeydh_info );
88 #endif
89 #if defined(MBEDTLS_ECDSA_C)
90  case MBEDTLS_PK_ECDSA:
91  return( &mbedtls_ecdsa_info );
92 #endif
93  /* MBEDTLS_PK_RSA_ALT omitted on purpose */
94  default:
95  return( NULL );
96  }
97 }

158 /*
159  * Helper for mbedtls_pk_sign and mbedtls_pk_verify
160  */
161 static inline int pk_hashlen_helper( mbedtls_md_type_t md_alg, size_t *hash_len )
162 {
163  const mbedtls_md_info_t *md_info;
164 
165  if( *hash_len != 0 )
166  return( 0 );
167 
168  if( ( md_info = mbedtls_md_info_from_type( md_alg ) ) == NULL )
169  return( -1 );
170 
171  *hash_len = mbedtls_md_get_size( md_info );
172  return( 0 );
173 }

175 /*
176  * Verify a signature
177  */
178 int mbedtls_pk_verify( mbedtls_pk_context *ctx, mbedtls_md_type_t md_alg,
179  const unsigned char *hash, size_t hash_len,
180  const unsigned char *sig, size_t sig_len )
181 {
182  if( ctx == NULL || ctx->pk_info == NULL ||
183  pk_hashlen_helper( md_alg, &hash_len ) != 0 )
184  return( MBEDTLS_ERR_PK_BAD_INPUT_DATA );
185 
186  if( ctx->pk_info->verify_func == NULL )
187  return( MBEDTLS_ERR_PK_TYPE_MISMATCH );
188 
189  return( ctx->pk_info->verify_func( ctx->pk_ctx, md_alg, hash, hash_len,
190  sig, sig_len ) );
191 }

193 /*
194  * Verify a signature with options
195  */
196 int mbedtls_pk_verify_ext( mbedtls_pk_type_t type, const void *options,
197  mbedtls_pk_context *ctx, mbedtls_md_type_t md_alg,
198  const unsigned char *hash, size_t hash_len,
199  const unsigned char *sig, size_t sig_len )
200 {
201  if( ctx == NULL || ctx->pk_info == NULL )
202  return( MBEDTLS_ERR_PK_BAD_INPUT_DATA );
203 
204  if( ! mbedtls_pk_can_do( ctx, type ) )
205  return( MBEDTLS_ERR_PK_TYPE_MISMATCH );
206 
207  if( type == MBEDTLS_PK_RSASSA_PSS )
208  {
209 #if defined(MBEDTLS_RSA_C) && defined(MBEDTLS_PKCS1_V21)
210  int ret;
211  const mbedtls_pk_rsassa_pss_options *pss_opts;
212 
213  if( options == NULL )
214  return( MBEDTLS_ERR_PK_BAD_INPUT_DATA );
215 
216  pss_opts = (const mbedtls_pk_rsassa_pss_options *) options;
217 
218  if( sig_len < mbedtls_pk_get_len( ctx ) )
219  return( MBEDTLS_ERR_RSA_VERIFY_FAILED );
220 
221  ret = mbedtls_rsa_rsassa_pss_verify_ext( mbedtls_pk_rsa( *ctx ),
222  NULL, NULL, MBEDTLS_RSA_PUBLIC,
223  md_alg, (unsigned int) hash_len, hash,
224  pss_opts->mgf1_hash_id,
225  pss_opts->expected_salt_len,
226  sig );
227  if( ret != 0 )
228  return( ret );
229 
230  if( sig_len > mbedtls_pk_get_len( ctx ) )
231  return( MBEDTLS_ERR_PK_SIG_LEN_MISMATCH );
232 
233  return( 0 );
234 #else
235  return( MBEDTLS_ERR_PK_FEATURE_UNAVAILABLE );
236 #endif
237  }
238 
239  /* General case: no options */
240  if( options != NULL )
241  return( MBEDTLS_ERR_PK_BAD_INPUT_DATA );
242 
243  return( mbedtls_pk_verify( ctx, md_alg, hash, hash_len, sig, sig_len ) );
244 }

301 /*
302  * Check public-private key pair
303  */
304 int mbedtls_pk_check_pair( const mbedtls_pk_context *pub, const mbedtls_pk_context *prv )
305 {
306  if( pub == NULL || pub->pk_info == NULL ||
307  prv == NULL || prv->pk_info == NULL ||
308  prv->pk_info->check_pair_func == NULL )
309  {
310  return( MBEDTLS_ERR_PK_BAD_INPUT_DATA );
311  }
312 
313  if( prv->pk_info->type == MBEDTLS_PK_RSA_ALT )
314  {
315  if( pub->pk_info->type != MBEDTLS_PK_RSA )
316  return( MBEDTLS_ERR_PK_TYPE_MISMATCH );
317  }
318  else
319  {
320  if( pub->pk_info != prv->pk_info )
321  return( MBEDTLS_ERR_PK_TYPE_MISMATCH );
322  }
323 
324  return( prv->pk_info->check_pair_func( pub->pk_ctx, prv->pk_ctx ) );
325 }
