48 static void mbedtls_zeroize( void *v, size_t n ) {
49  volatile unsigned char *p = v; while( n-- ) *p++ = 0;
50 }

52 /*
53  * If DTLS is in use, then at least one of SHA-1, SHA-256, SHA-512 is
54  * available. Try SHA-256 first, 512 wastes resources since we need to stay
55  * with max 32 bytes of cookie for DTLS 1.0
56  */
57 #if defined(MBEDTLS_SHA256_C)
58 #define COOKIE_MD MBEDTLS_MD_SHA224
59 #define COOKIE_MD_OUTLEN 32
60 #define COOKIE_HMAC_LEN 28
61 #elif defined(MBEDTLS_SHA512_C)
62 #define COOKIE_MD MBEDTLS_MD_SHA384
63 #define COOKIE_MD_OUTLEN 48
64 #define COOKIE_HMAC_LEN 28
65 #elif defined(MBEDTLS_SHA1_C)
66 #define COOKIE_MD MBEDTLS_MD_SHA1
67 #define COOKIE_MD_OUTLEN 20
68 #define COOKIE_HMAC_LEN 20
69 #else
70 #error "DTLS hello verify needs SHA-1 or SHA-2"
71 #endif

73 /*
74  * Cookies are formed of a 4-bytes timestamp (or serial number) and
75  * an HMAC of timestemp and client ID.
76  */
77 #define COOKIE_LEN ( 4 + COOKIE_HMAC_LEN )

108 int mbedtls_ssl_cookie_setup( mbedtls_ssl_cookie_ctx *ctx,
109  int (*f_rng)(void *, unsigned char *, size_t),
110  void *p_rng )
111 {
112  int ret;
113  unsigned char key[COOKIE_MD_OUTLEN];
114 
115  if( ( ret = f_rng( p_rng, key, sizeof( key ) ) ) != 0 )
116  return( ret );
117 
118  ret = mbedtls_md_setup( &ctx->hmac_ctx, mbedtls_md_info_from_type( COOKIE_MD ), 1 );
119  if( ret != 0 )
120  return( ret );
121 
122  ret = mbedtls_md_hmac_starts( &ctx->hmac_ctx, key, sizeof( key ) );
123  if( ret != 0 )
124  return( ret );
125 
126  mbedtls_zeroize( key, sizeof( key ) );
127 
128  return( 0 );
129 }

131 /*
132  * Generate the HMAC part of a cookie
133  */
134 static int ssl_cookie_hmac( mbedtls_md_context_t *hmac_ctx,
135  const unsigned char time[4],
136  unsigned char **p, unsigned char *end,
137  const unsigned char *cli_id, size_t cli_id_len )
138 {
139  unsigned char hmac_out[COOKIE_MD_OUTLEN];
140 
141  if( (size_t)( end - *p ) < COOKIE_HMAC_LEN )
142  return( MBEDTLS_ERR_SSL_BUFFER_TOO_SMALL );
143 
144  if( mbedtls_md_hmac_reset( hmac_ctx ) != 0 ||
145  mbedtls_md_hmac_update( hmac_ctx, time, 4 ) != 0 ||
146  mbedtls_md_hmac_update( hmac_ctx, cli_id, cli_id_len ) != 0 ||
147  mbedtls_md_hmac_finish( hmac_ctx, hmac_out ) != 0 )
148  {
149  return( MBEDTLS_ERR_SSL_INTERNAL_ERROR );
150  }
151 
152  memcpy( *p, hmac_out, COOKIE_HMAC_LEN );
153  *p += COOKIE_HMAC_LEN;
154 
155  return( 0 );
156 }

158 /*
159  * Generate cookie for DTLS ClientHello verification
160  */
161 int mbedtls_ssl_cookie_write( void *p_ctx,
162  unsigned char **p, unsigned char *end,
163  const unsigned char *cli_id, size_t cli_id_len )
164 {
165  int ret;
166  mbedtls_ssl_cookie_ctx *ctx = (mbedtls_ssl_cookie_ctx *) p_ctx;
167  unsigned long t;
168 
169  if( ctx == NULL || cli_id == NULL )
170  return( MBEDTLS_ERR_SSL_BAD_INPUT_DATA );
171 
172  if( (size_t)( end - *p ) < COOKIE_LEN )
173  return( MBEDTLS_ERR_SSL_BUFFER_TOO_SMALL );
174 
175 #if defined(MBEDTLS_HAVE_TIME)
176  t = (unsigned long) time( NULL );
177 #else
178  t = ctx->serial++;
179 #endif
180 
181  (*p)[0] = (unsigned char)( t >> 24 );
182  (*p)[1] = (unsigned char)( t >> 16 );
183  (*p)[2] = (unsigned char)( t >> 8 );
184  (*p)[3] = (unsigned char)( t );
185  *p += 4;
186 
187 #if defined(MBEDTLS_THREADING_C)
188  if( ( ret = mbedtls_mutex_lock( &ctx->mutex ) ) != 0 )
189  return( MBEDTLS_ERR_SSL_INTERNAL_ERROR + ret );
190 #endif
191 
192  ret = ssl_cookie_hmac( &ctx->hmac_ctx, *p - 4,
193  p, end, cli_id, cli_id_len );
194 
195 #if defined(MBEDTLS_THREADING_C)
196  if( mbedtls_mutex_unlock( &ctx->mutex ) != 0 )
197  return( MBEDTLS_ERR_SSL_INTERNAL_ERROR +
198  MBEDTLS_ERR_THREADING_MUTEX_ERROR );
199 #endif
200 
201  return( ret );
202 }

204 /*
205  * Check a cookie
206  */
207 int mbedtls_ssl_cookie_check( void *p_ctx,
208  const unsigned char *cookie, size_t cookie_len,
209  const unsigned char *cli_id, size_t cli_id_len )
210 {
211  unsigned char ref_hmac[COOKIE_HMAC_LEN];
212  int ret = 0;
213  unsigned char *p = ref_hmac;
214  mbedtls_ssl_cookie_ctx *ctx = (mbedtls_ssl_cookie_ctx *) p_ctx;
215  unsigned long cur_time, cookie_time;
216 
217  if( ctx == NULL || cli_id == NULL )
218  return( MBEDTLS_ERR_SSL_BAD_INPUT_DATA );
219 
220  if( cookie_len != COOKIE_LEN )
221  return( -1 );
222 
223 #if defined(MBEDTLS_THREADING_C)
224  if( ( ret = mbedtls_mutex_lock( &ctx->mutex ) ) != 0 )
225  return( MBEDTLS_ERR_SSL_INTERNAL_ERROR + ret );
226 #endif
227 
228  if( ssl_cookie_hmac( &ctx->hmac_ctx, cookie,
229  &p, p + sizeof( ref_hmac ),
230  cli_id, cli_id_len ) != 0 )
231  ret = -1;
232 
233 #if defined(MBEDTLS_THREADING_C)
234  if( mbedtls_mutex_unlock( &ctx->mutex ) != 0 )
235  return( MBEDTLS_ERR_SSL_INTERNAL_ERROR +
236  MBEDTLS_ERR_THREADING_MUTEX_ERROR );
237 #endif
238 
239  if( ret != 0 )
240  return( ret );
241 
242  if( mbedtls_ssl_safer_memcmp( cookie + 4, ref_hmac, sizeof( ref_hmac ) ) != 0 )
243  return( -1 );
244 
245 #if defined(MBEDTLS_HAVE_TIME)
246  cur_time = (unsigned long) time( NULL );
247 #else
248  cur_time = ctx->serial;
249 #endif
250 
251  cookie_time = ( (unsigned long) cookie[0] << 24 ) |
252  ( (unsigned long) cookie[1] << 16 ) |
253  ( (unsigned long) cookie[2] << 8 ) |
254  ( (unsigned long) cookie[3] );
255 
256  if( ctx->timeout != 0 && cur_time - cookie_time > ctx->timeout )
257  return( -1 );
258 
259  return( 0 );
260 }
