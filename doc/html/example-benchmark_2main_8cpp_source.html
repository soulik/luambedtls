295 static int myrand( void *rng_state, unsigned char *output, size_t len )
296 {
297  size_t use_len;
298  int rnd;
299 
300  if( rng_state != NULL )
301  rng_state = NULL;
302 
303  while( len > 0 )
304  {
305  use_len = len;
306  if( use_len > sizeof(int) )
307  use_len = sizeof(int);
308 
309  rnd = rand();
310  memcpy( output, &rnd, use_len );
311  output += use_len;
312  len -= use_len;
313  }
314 
315  return( 0 );
316 }
